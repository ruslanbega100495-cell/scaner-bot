Да, можно одной командой! 🚀
📦 Создай файл start.bat (в корне проекта)
batch

@echo off
echo 🚀 Запуск Freelance Monitor Bot...

REM Проверка Docker
docker-compose -p scanner-bot ps >nul 2>&1
if %errorlevel% neq 0 (
    echo ⚙️  Запускаю контейнеры...
    docker-compose -p scanner-bot up -d
)

REM Проверка зависимостей
if not exist node_modules (
    echo ⚙️  Устанавливаю зависимости...
    npm install
)

REM Запуск приложения
echo 🔄 Запускаю приложение...
npm run dev

pause



🔧 Как использовать:
Сохрани как start.bat в папке проекта
Запусти двойным кликом ✅
🎯 Быстрые команды (если вручную)

🛑 Остановка
powershell


# Остановить приложение: Ctrl+C в терминале

# Остановить контейнеры:
docker-compose -p scanner-bot down

# Или одной командой всё:
taskkill /F /IM node.exe 2>$null && docker-compose -p scanner-bot down



🔍 Проверка работы

# Сервер жив?
