@echo off
echo 🚀 Запуск Freelance Monitor Bot...

REM Проверка Docker
docker-compose -p scanner-bot ps >nul 2>&1
if %errorlevel% neq 0 (
    echo ⚙️  Запускаю контейнеры...
    docker-compose -p scanner-bot up -d
)

REM Проверка зависимостей
if not exist node_modules (
    echo ⚙️  Устанавливаю зависимости...
    npm install
)

REM Запуск приложения
echo 🔄 Запускаю приложение...
npm run dev

pause


 Проверка работы


# Сервер жив?
curl.exe http://localhost:3000/health

# Запустить скрапинг:
curl.exe "http://localhost:3000/api/v1/jobs/scan?source=kwork"

# Посмотреть заказы:
curl.exe http://localhost:3000/api/v1/jobs

# Смотреть логи:
Get-Content logs/all.log -Tail 20 -Wait
